from sklearn.linear_model import LinearRegression
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import PolynomialFeatures

xfit = np.linspace(0, 4*np.pi, 1000)

####################         YOUR CODE         ####################
degree=[3, 5, 9, 15]
fig, axs = plt.subplots(2, 2)

idx = 0
sec = 0

for deg in degree:
  estimator = make_pipeline(PolynomialFeatures(deg), LinearRegression())

  estimator.fit(X[:, np.newaxis], y[:, np.newaxis])
  yfit = estimator.predict(xfit[:, np.newaxis])

  if deg < 6:
    axs[0, idx].plot(xfit, yfit, 'r')
    axs[0, idx].scatter(X, y)
    axs[0, idx].set_title("Polynomial Fit with Degree " + str(deg))
    idx += 1
  if deg > 6:
    axs[1, sec].plot(xfit, yfit, 'r')
    axs[1, sec].scatter(X, y)
    axs[1, sec].set_title("Polynomial Fit with Degree " + str(deg))
    sec += 1
